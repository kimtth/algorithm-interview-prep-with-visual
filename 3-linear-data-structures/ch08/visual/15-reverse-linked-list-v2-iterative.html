Reverse a singly linked list.

def reverseList(self, head: ListNode) -> ListNode:
    node, prev = head, None
    
    while node:
        next, node.next = node.next, prev  # Save next, reverse link
        prev, node = node, next            # Move pointers forward
    
    return prev
