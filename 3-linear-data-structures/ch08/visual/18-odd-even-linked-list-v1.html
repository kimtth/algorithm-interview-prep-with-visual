Q18: Odd Even Linked List

Group all odd-indexed nodes together followed by even-indexed nodes. Index starts at 1.

üíª Python Solution

def oddEvenList(self, head: ListNode) -> ListNode:
    if head is None:
        return None

    odd = head
    even = head.next
    even_head = head.next  # Save even list head

    # Process odd and even nodes
    while even and even.next:
        odd.next, even.next = odd.next.next, even.next.next
        odd, even = odd.next, even.next

    # Connect odd tail to even head
    odd.next = even_head
    return head
