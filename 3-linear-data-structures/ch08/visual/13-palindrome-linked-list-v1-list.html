Q13: Palindrome Linked List

Given the head of a singly linked list, determine if it is a palindrome (reads the same forward and backward).

def isPalindrome(self, head: ListNode) -> bool:
    q: List = []
    
    if not head:
        return True
    
    node = head
    # Convert to list
    while node is not None:
        q.append(node.val)
        node = node.next
    
    # Check palindrome
    while len(q) > 1:
        if q.pop(0) != q.pop():
            return False
    
    return True
