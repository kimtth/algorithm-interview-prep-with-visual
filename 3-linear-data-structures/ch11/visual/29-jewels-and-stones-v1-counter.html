Q29: Jewels and Stones

Count how many stones you have that are also jewels.

def numJewelsInStones(self, J: str, S: str) -> int:
    freqs = collections.Counter(S)  # Count stone frequencies
    count = 0
    for char in J:
        count += freqs[char]  # Sum jewel counts
    return count
