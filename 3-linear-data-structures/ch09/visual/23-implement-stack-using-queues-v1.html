Q23: Implement Stack using Queues

Implement a LIFO stack using only FIFO queue operations.

üíª Python Solution

class MyStack:
    def __init__(self):
        self.q = collections.deque()

    def push(self, x):
        self.q.append(x)
        # Rotate to put new element at front
        for _ in range(len(self.q) - 1):
            self.q.append(self.q.popleft())

    def pop(self):
        return self.q.popleft()

    def top(self):
        return self.q[0]
