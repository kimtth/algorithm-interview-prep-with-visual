Q21: Remove Duplicate Letters

Remove duplicate letters so that every letter appears once and result is the smallest in lexicographical order.

def removeDuplicateLetters(self, s: str) -> str:
    # For each char in sorted unique chars
    for char in sorted(set(s)):
        suffix = s[s.index(char):]
        # If suffix contains all unique chars, proceed
        if set(s) == set(suffix):
            return char + self.removeDuplicateLetters(
                suffix.replace(char, ''))
    return ''
