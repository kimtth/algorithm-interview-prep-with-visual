Q20: Valid Parentheses

Determine if the input string has valid matching brackets: (), {}, []

def isValid(self, s: str) -> bool:
    stack = []
    table = {
        ')': '(',
        '}': '{',
        ']': '[',
    }

    for char in s:
        if char not in table:
            stack.append(char)  # Opening bracket
        elif not stack or table[char] != stack.pop():
            return False
    return len(stack) == 0
