Q22: Daily Temperatures

Given temperatures, return days until a warmer temperature. Return 0 if no warmer day.

def dailyTemperatures(self, T: List[int]) -> List[int]:
    answer = [0] * len(T)
    stack = []
    for i, cur in enumerate(T):
        # Pop while current temp > stack top
        while stack and cur > T[stack[-1]]:
            last = stack.pop()
            answer[last] = i - last
        stack.append(i)
    return answer
