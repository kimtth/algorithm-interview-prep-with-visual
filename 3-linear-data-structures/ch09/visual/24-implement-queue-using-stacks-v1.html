Q24: Implement Queue using Stacks

Implement a FIFO queue using only LIFO stack operations.

üíª Python Solution

class MyQueue:
    def __init__(self):
        self.input = []
        self.output = []

    def push(self, x):
        self.input.append(x)

    def pop(self):
        self.peek()
        return self.output.pop()

    def peek(self):
        # Transfer input to output if output empty
        if not self.output:
            while self.input:
                self.output.append(self.input.pop())
        return self.output[-1]
