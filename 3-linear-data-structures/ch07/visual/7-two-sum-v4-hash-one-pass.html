def twoSum(self, nums: List[int], target: int) -> List[int]:
    nums_map = {}
    # Build map and search in single loop
    for i, num in enumerate(nums):
        if target - num in nums_map:
            return [nums_map[target - num], i]
        nums_map[num] = i
