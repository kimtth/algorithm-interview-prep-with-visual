Given two integer arrays nums1 and nums2, return an array of their intersection. Each element must appear as many times as it shows in both arrays.

Key Insight: Sort both arrays, use two pointers. If nums1[i] < nums2[j], advance i; if nums1[i] > nums2[j], advance j; if equal, add to result and advance both.

üìÑ Python Solution (67-3.py)

from typing import List, Set

class Solution:
    def intersection(self, nums1: List[int], nums2: List[int]) -> List[int]:
        result: Set = set()
        nums1.sort()
        nums2.sort()
        i = j = 0
        while i < len(nums1) and j < len(nums2):
            if nums1[i] > nums2[j]:
                j += 1
            elif nums1[i] < nums2[j]:
                i += 1
            else:
                result.add(nums1[i])
                i += 1
                j += 1
        return result
