Maximize loot without robbing adjacent houses.

Key Insight: Use OrderedDict to maintain only the last 2 DP values. dp[i] = max(dp[i-1], dp[i-2] + nums[i]). Pop oldest entry to save space.

üìÑ Python Solution (88-2.py)

import collections
from typing import List

class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        if len(nums) <= 2:
            return max(nums)

        dp = collections.OrderedDict()
        dp[0], dp[1] = nums[0], max(nums[0], nums[1])
        for i in range(2, len(nums)):
            dp[i] = max(dp[i - 1], dp[i - 2] + nums[i])
            dp.popitem(last=False)  # Remove oldest entry
        return dp.popitem()[1]
