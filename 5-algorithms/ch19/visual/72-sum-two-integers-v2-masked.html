Calculate the sum of two integers without using + or - operators.

Key Insight: Use XOR for sum without carry, AND+shift for carry. Apply 32-bit mask (0xFFFFFFFF) for Python's infinite precision. Handle negative overflow with INT_MAX check.

class Solution:
    def getSum(self, a: int, b: int) -> int:
        MASK = 0xFFFFFFFF
        INT_MAX = 0x7FFFFFFF
        # Sum and carry
        while b != 0:
            a, b = (a ^ b) & MASK, ((a & b) << 1) & MASK

        # Handle negative result
        if a > INT_MAX:
            a = ~(a ^ MASK)
        return a
