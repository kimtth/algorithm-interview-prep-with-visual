Key Insight: Sort both arrays. For each cookie, use bisect_right to find how many children can be satisfied. If more than current result, increment.

import bisect
from typing import List

class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        g.sort()
        s.sort()
        result = 0
        for i in s:
            # Find how many children can be satisfied
            index = bisect.bisect_right(g, i)
            if index > result:
                result += 1
        return result
