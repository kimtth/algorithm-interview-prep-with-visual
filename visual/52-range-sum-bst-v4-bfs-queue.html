# Definition for a binary tree node.
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None


class Solution:
    def rangeSumBST(self, root: TreeNode, L: int, R: int) -> int:
        stack, sum = [root], 0
        # ÌÅê Ïó∞ÏÇ∞ÏùÑ Ïù¥Ïö©Ìï¥ Î∞òÎ≥µ Íµ¨Ï°∞ BFSÎ°ú ÌïÑÏöîÌïú ÎÖ∏Îìú ÌÉêÏÉâ
        ÏúÑÏóê ÏÑ§Î™ÖÍ≥º ÎßàÏ∞¨Í∞ÄÏßÄ
        while stack:
            node = stack.pop(0)
            if node:
                if node.val > L:
                    stack.append(node.left)
                if node.val < R:
                    stack.append(node.right)
                if L <= node.val <= R:
                    sum += node.val
        return sum
