from typing import List


class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        results = []
        prev_elements = []

        def dfs(elements):
            # Î¶¨ÌîÑ ÎÖ∏ÎìúÏùºÎïå Í≤∞Í≥º Ï∂îÍ∞Ä
            if len(elements) == 0:
                results.append(prev_elements[:])

            # ÏàúÏó¥ ÏÉùÏÑ± Ïû¨Í∑Ä Ìò∏Ï∂ú
            for e in elements:
                next_elements = elements[:]
                next_elements.remove(e)

                prev_elements.append(e)
                dfs(next_elements)
                prev_elements.pop()

        dfs(nums)
        return results
