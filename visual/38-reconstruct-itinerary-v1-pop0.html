import collections
from typing import List


class Solution:
    def findItinerary(self, tickets: List[List[str]]) -> List[str]:
        graph = collections.defaultdict(list)
        # Í∑∏ÎûòÌîÑ ÏàúÏÑúÎåÄÎ°ú Íµ¨ÏÑ±
        for a, b in sorted(tickets):
            graph[a].append(b)

        route = []

        def dfs(a):
            # Ï≤´ Î≤àÏß∏ Í∞íÏùÑ ÏùΩÏñ¥ Ïñ¥ÌúòÏàú Î∞©Î¨∏
            while graph[a]:
                dfs(graph[a].pop(0))
            route.append(a)

        dfs('JFK')
        # Îã§Ïãú Îí§ÏßëÏñ¥ Ïñ¥ÌúòÏàú Í≤∞Í≥ºÎ°ú
        return route[::-1]
