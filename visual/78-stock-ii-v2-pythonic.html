from typing import List


class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        # 0Î≥¥Îã§ ÌÅ¨Î©¥ Î¨¥Ï°∞Í±¥ Ìï©ÏÇ∞
        return sum(max(prices[i + 1] - prices[i], 0) for i in range(len(prices) - 1))
