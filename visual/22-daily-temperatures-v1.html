from typing import List


class Solution:
    def dailyTemperatures(self, T: List[int]) -> List[int]:
        answer = [0] * len(T)
        stack = []
        for i, cur in enumerate(T):
            # í˜„ì¬ ì˜¨ë„ê°€ ìŠ¤íƒ ê°’ë³´ë‹¤ ë†’ë‹¤ë©´ ì •ë‹µ ì²˜ë¦¬
            while stack and cur > T[stack[-1]]:
                last = stack.pop()
                answer[last] = i - last
            stack.append(i)

        return answer
