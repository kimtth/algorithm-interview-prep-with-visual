from typing import List


class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        result = []

        def dfs(index, path):
            # Îß§ Î≤à Í≤∞Í≥º Ï∂îÍ∞Ä
            result.append(path)

            # Í≤ΩÎ°úÎ•º ÎßåÎì§Î©¥ÏÑú DFS
            for i in range(index, len(nums)):
                dfs(i + 1, path + [nums[i]])

        dfs(0, [])
        return result
