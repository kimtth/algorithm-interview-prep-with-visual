import collections
import heapq
from typing import List


class Solution:
    def networkDelayTime(self, times: List[List[int]], N: int, K: int) -> int:
        graph = collections.defaultdict(list)
        # ê·¸ë˜í”„ ì¸ì ‘ ë¦¬ìŠ¤íŠ¸ êµ¬ì„±
        for u, v, w in times:
            graph[u].append((v, w))

        # í ë³€ìˆ˜: [(ì†Œìš” ì‹œê°„, ì •ì )]
        Q = [(0, K)]
        dist = collections.defaultdict(int)

        # ìš°ì„  ìˆœìœ„ í ìµœì†Œê°’ ê¸°ì¤€ìœ¼ë¡œ ì •ì ê¹Œì§€ ìµœë‹¨ ê²½ë¡œ ì‚½ì…
        while Q:
            time, node = heapq.heappop(Q)
            if node not in dist:
                dist[node] = time
                for v, w in graph[node]:
                    alt = time + w
                    heapq.heappush(Q, (alt, v))

        # ëª¨ë“  ë…¸ë“œ ìµœë‹¨ ê²½ë¡œ ì¡´ì¬ ì—¬ë¶€ íŒë³„
        if len(dist) == N:
            return max(dist.values())
        return -1
