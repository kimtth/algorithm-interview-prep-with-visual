# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None


class Solution:
    def reverseBetween(self, head: ListNode, m: int, n: int) -> ListNode:

        if not head or m == n:
            return head

        root = start = ListNode(None)
        root.next = head

        return root.next
