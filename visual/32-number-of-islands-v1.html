from typing import List


class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        def dfs(i, j):
            # Îçî Ïù¥ÏÉÅ ÎïÖÏù¥ ÏïÑÎãå Í≤ΩÏö∞ Ï¢ÖÎ£å
            if i < 0 or i >= len(grid) or \
                    j < 0 or j >= len(grid[0]) or \
                    grid[i][j] != '1':
                return

            grid[i][j] = 0

            # ÎèôÏÑúÎÇ®Î∂Å ÌÉêÏÉâ
            dfs(i + 1, j)
            dfs(i - 1, j)
            dfs(i, j + 1)
            dfs(i, j - 1)

        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == '1':
                    dfs(i, j)
                    # Î™®Îì† Ïú°ÏßÄ ÌÉêÏÉâ ÌõÑ Ïπ¥Ïö¥Ìä∏ 1 Ï¶ùÍ∞Ä
                    count += 1
        return count
