# Definition for a binary tree node.
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None


class Solution:
    def isBalanced(self, root: TreeNode) -> bool:
        def check(root):
            if not root:
                return 0

            left = check(root.left)
            right = check(root.right)
            # ÎÜíÏù¥ Ï∞®Ïù¥Í∞Ä ÎÇòÎäî Í≤ΩÏö∞ -1, Ïù¥Ïô∏ÏóêÎäî ÎÜíÏù¥Ïóê Îî∞Îùº 1 Ï¶ùÍ∞Ä
            if left == -1 or \
                    right == -1 or \
                    abs(left - right) > 1:
                return -1
            return max(left, right) + 1

        return check(root) != -1
