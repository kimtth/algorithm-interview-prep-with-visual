# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None


class Solution:
    def swapPairs(self, head: ListNode) -> ListNode:
        if head and head.next:
            p = head.next
            # Ïä§ÏôëÎêú Í∞í Î¶¨ÌÑ¥ Î∞õÏùå
            head.next = self.swapPairs(p.next)
            p.next = head
            return p
        return head
