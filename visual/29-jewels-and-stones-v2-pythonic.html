import collections


class Solution:
    def numJewelsInStones(self, J: str, S: str) -> int:
        freqs = collections.defaultdict(int)
        count = 0

        # ÎπÑÍµê ÏóÜÏù¥ Îèå(S) ÎπàÎèÑ Ïàò Í≥ÑÏÇ∞
        for char in S:
            freqs[char] += 1

        # ÎπÑÍµê ÏóÜÏù¥ Î≥¥ÏÑù(J) ÎπàÎèÑ Ïàò Ìï©ÏÇ∞
        for char in J:
            count += freqs[char]

        return count
