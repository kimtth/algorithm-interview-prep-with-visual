from typing import List


class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        g.sort()
        s.sort()

        child_i = cookie_j = 0
        # ë§Œì¡±í•˜ì§€ ëª»í•  ë•Œê¹Œì§€ ê·¸ë¦¬ë”” ì§„í–‰
        while child_i < len(g) and cookie_j < len(s):
            if s[cookie_j] >= g[child_i]:
                child_i += 1
            cookie_j += 1

        return child_i
