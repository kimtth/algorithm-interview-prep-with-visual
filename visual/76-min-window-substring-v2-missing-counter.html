Find the minimum window in s that contains all characters of t.

Key Insight: Use a "missing" counter to track how many chars still needed. When missing=0, we have a valid window. Shrink from left while valid.

import collections


class Solution:
    def minWindow(self, s: str, t: str) -> str:
        need = collections.Counter(t)
        missing = len(t)
        left = start = end = 0

        # Ïò§Î•∏Ï™Ω Ìè¨Ïù∏ÌÑ∞ Ïù¥Îèô
        for right, char in enumerate(s, 1):
            missing -= need[char] > 0
            need[char] -= 1

            # ÌïÑÏöî Î¨∏ÏûêÍ∞Ä 0Ïù¥Î©¥ ÏôºÏ™Ω Ìè¨Ïù∏ÌÑ∞ Ïù¥Îèô ÌåêÎã®
            if missing == 0:
                while left < right and need[s[left]] < 0:
                    need[s[left]] += 1
                    left += 1

                if not end or right - left <= end - start:
                    start, end = left, right
                need[s[left]] += 1
                missing += 1
                left += 1
        return s[start:end]
