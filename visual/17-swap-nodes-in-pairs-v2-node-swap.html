# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None


class Solution:
    def swapPairs(self, head: ListNode) -> ListNode:
        root = prev = ListNode(None)
        prev.next = head
        while head and head.next:
            # bÍ∞Ä a(head)Î•º Í∞ÄÎ¶¨ÌÇ§ÎèÑÎ°ù
            b = head.next
            head.next = b.next
            b.next = head

            # prevÍ∞Ä bÎ•º Í∞ÄÎ¶¨ÌÇ§ÎèÑÎ°ù
            prev.next = b

            # Îã§Ïùå Î≤à ÎπÑÍµêÎ•º ÏúÑÌï¥ Ïù¥Îèô
            head = head.next
            prev = prev.next.next
        return root.next
