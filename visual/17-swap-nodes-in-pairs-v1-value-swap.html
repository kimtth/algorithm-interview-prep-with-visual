# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None


class Solution:
    def swapPairs(self, head: ListNode) -> ListNode:
        cur = head

        while cur and cur.next:
            # Í∞íÎßå ÍµêÌôò
            cur.val, cur.next.val = cur.next.val, cur.val
            cur = cur.next.next

        return head
