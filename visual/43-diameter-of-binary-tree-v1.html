# Definition for a binary tree node.
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None


class Solution:
    longest: int = 0

    def diameterOfBinaryTree(self, root: TreeNode) -> int:
        def dfs(node: TreeNode) -> int:
            if not node:
                return -1
            # ÏôºÏ™Ω, Ïò§Î•∏Ï™Ω Í∞ÅÍ∞Å Î¶¨ÌîÑ ÎÖ∏ÎìúÍπåÏßÄ ÌÉêÏÉâ
            left = dfs(node.left)
            right = dfs(node.right)

            # Í∞ÄÏû• Í∏¥ Í≤ΩÎ°ú
            self.longest = max(self.longest, left + right + 2)
            # ÏÉÅÌÉúÍ∞í
            return max(left, right) + 1

        dfs(root)
        return self.longest
