import collections
import heapq
from typing import List


class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        freqs = collections.Counter(nums)
        freqs_heap = []
        # í™ì— ìŒìˆ˜ë¡œ ì‚½ì…
        for f in freqs:
            heapq.heappush(freqs_heap, (-freqs[f], f))
