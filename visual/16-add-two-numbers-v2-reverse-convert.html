# Definition for singly-linked list.
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None


class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        root = head = ListNode(0)

        carry = 0
        while l1 or l2 or carry:
            sum = 0
            # Îëê ÏûÖÎ†•Í∞íÏùò Ìï© Í≥ÑÏÇ∞
            if l1:
                sum += l1.val
                l1 = l1.next
            if l2:
                sum += l2.val
                l2 = l2.next

            # Î™´(ÏûêÎ¶¨Ïò¨Î¶ºÏàò)Í≥º ÎÇòÎ®∏ÏßÄ(Í∞í) Í≥ÑÏÇ∞
            carry, val = divmod(sum + carry, 10)
            head.next = ListNode(val)
            head = head.next

        return root.next
