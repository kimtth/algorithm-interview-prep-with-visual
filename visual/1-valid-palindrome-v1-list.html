class Solution:
    def isPalindrome(self, s: str) -> bool:
        strs = []
        for char in s:
            if char.isalnum():
                strs.append(char.lower())

        # íŒ°ë¦°ë“œë¡¬ ì—¬ë¶€ íŒë³„
        while len(strs) > 1:
            if strs.pop(0) != strs.pop():
                return False

        return True
