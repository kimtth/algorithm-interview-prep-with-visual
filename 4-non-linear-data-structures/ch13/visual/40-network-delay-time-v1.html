Q40: Network Delay Time

Given a network of n nodes, find the time it takes for a signal to reach all nodes from source K using Dijkstra's algorithm.

üíª Python Solution

def networkDelayTime(times: List[List[int]], N: int, K: int) -> int:
    graph = collections.defaultdict(list)
    for u, v, w in times:
        graph[u].append((v, w))
    
    Q = [(0, K)]  # (time, node)
    dist = collections.defaultdict(int)
    
    while Q:
        time, node = heapq.heappop(Q)
        if node not in dist:
            dist[node] = time
            for v, w in graph[node]:
                alt = time + w
                heapq.heappush(Q, (alt, v))
    
    if len(dist) == N:
        return max(dist.values())
    return -1
