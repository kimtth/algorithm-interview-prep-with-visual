Q38: Reconstruct Itinerary - DFS pop()

Optimized version using reverse sort + pop() instead of pop(0) for O(1) removal.

def findItinerary(tickets: List[List[str]]) -> List[str]:
    graph = collections.defaultdict(list)
    # Build graph in REVERSE sorted order
    for a, b in sorted(tickets, reverse=True):
        graph[a].append(b)

    route = []

    def dfs(a):
        while graph[a]:
            dfs(graph[a].pop())  # pop() is O(1) vs pop(0) is O(n)
        route.append(a)

    dfs('JFK')
    return route[::-1]
