Optimized version with visited set to avoid revisiting already-verified nodes.

def canFinish(numCourses: int, prerequisites: List[List[int]]) -> bool:
    graph = collections.defaultdict(list)
    for x, y in prerequisites:
        graph[x].append(y)

    traced = set()   # Currently in DFS path
    visited = set()  # Already verified (no cycle from here)

    def dfs(i):
        if i in traced:
            return False
        if i in visited:
            return True  # Skip - already verified!

        traced.add(i)
        for y in graph[i]:
            if not dfs(y):
                return False
        traced.remove(i)
        visited.add(i)  # Mark as verified
        return True

    for x in list(graph):
        if not dfs(x):
            return False
    return True
