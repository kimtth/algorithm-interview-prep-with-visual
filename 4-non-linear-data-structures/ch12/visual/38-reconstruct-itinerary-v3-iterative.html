Q38: Reconstruct Itinerary - Iterative Stack

Non-recursive solution using explicit stack to avoid potential stack overflow.

def findItinerary(tickets: List[List[str]]) -> List[str]:
    graph = collections.defaultdict(list)
    for a, b in sorted(tickets):
        graph[a].append(b)

    route, stack = [], ['JFK']
    while stack:
        # Keep going while current node has outgoing edges
        while graph[stack[-1]]:
            stack.append(graph[stack[-1]].pop(0))
        # No more edges, add to route
        route.append(stack.pop())

    return route[::-1]
