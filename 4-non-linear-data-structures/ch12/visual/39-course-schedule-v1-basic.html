Q39: Course Schedule - Basic DFS

Determine if you can finish all courses given prerequisites. Detect if there's a cycle in the dependency graph.

üíª Python Solution (Basic DFS)

def canFinish(numCourses: int, prerequisites: List[List[int]]) -> bool:
    graph = collections.defaultdict(list)
    for x, y in prerequisites:
        graph[x].append(y)

    traced = set()  # Currently in DFS path

    def dfs(i):
        if i in traced:
            return False  # Cycle detected!

        traced.add(i)
        for y in graph[i]:
            if not dfs(y):
                return False
        traced.remove(i)  # Backtrack
        return True

    for x in list(graph):
        if not dfs(x):
            return False
    return True
