Q34: Permutations - DFS

Given an array of distinct integers, return all possible permutations using DFS backtracking.

üíª Python Solution

def permute(nums: List[int]) -> List[List[int]]:
    results = []
    prev_elements = []

    def dfs(elements):
        if len(elements) == 0:
            results.append(prev_elements[:])  # Leaf node
            return

        for e in elements:
            next_elements = elements[:]
            next_elements.remove(e)

            prev_elements.append(e)
            dfs(next_elements)
            prev_elements.pop()  # Backtrack

    dfs(nums)
    return results
