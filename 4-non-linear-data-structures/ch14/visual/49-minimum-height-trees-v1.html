Q49: Minimum Height Trees

Find roots that minimize tree height by repeatedly removing leaf nodes until 1-2 nodes remain.

üíª Python Solution

def findMinHeightTrees(n: int, edges: List) -> List[int]:
    if n <= 1:
        return [0]
    
    graph = collections.defaultdict(list)
    for i, j in edges:
        graph[i].append(j)
        graph[j].append(i)
    
    # Find initial leaves (degree = 1)
    leaves = [i for i in range(n) if len(graph[i]) == 1]
    
    # Trim leaves until 1-2 nodes remain
    while n > 2:
        n -= len(leaves)
        new_leaves = []
        for leaf in leaves:
            neighbor = graph[leaf].pop()
            graph[neighbor].remove(leaf)
            if len(graph[neighbor]) == 1:
                new_leaves.append(neighbor)
        leaves = new_leaves
    
    return leaves
