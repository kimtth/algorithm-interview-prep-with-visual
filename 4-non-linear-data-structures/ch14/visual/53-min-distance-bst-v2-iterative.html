Q53: Minimum Distance in BST (v2 - Iterative)

Find minimum difference using iterative in-order traversal with explicit stack. Morris traversal alternative.

def minDiffInBST(root: TreeNode) -> int:
    prev = -sys.maxsize
    result = sys.maxsize

    stack = []
    node = root

    # Î∞òÎ≥µ Íµ¨Ï°∞ Ï§ëÏúÑ ÏàúÌöå ÎπÑÍµê Í≤∞Í≥º
    while stack or node:
        while node:              # Go left as far as possible
            stack.append(node)
            node = node.left

        node = stack.pop()      # Process node
        result = min(result, node.val - prev)
        prev = node.val

        node = node.right       # Move to right subtree

    return result
