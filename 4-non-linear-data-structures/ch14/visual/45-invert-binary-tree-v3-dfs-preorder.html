def invertTree(root: TreeNode) -> TreeNode:
    stack = collections.deque([root])

    while stack:
        node = stack.pop()
        # Î∂ÄÎ™® ÎÖ∏Îìú Î∂ÄÌÑ∞ ÌïòÌñ•Ïãù Ïä§Ïôë
        if node:
            node.left, node.right = node.right, node.left

            stack.append(node.left)
            stack.append(node.right)

    return root
