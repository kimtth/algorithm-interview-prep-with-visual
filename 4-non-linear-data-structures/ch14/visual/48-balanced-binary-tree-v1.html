Q48: Balanced Binary Tree

Check if a binary tree is height-balanced (heights of subtrees differ by at most 1).

üíª Python Solution

def isBalanced(root: TreeNode) -> bool:
    def check(root):
        if not root:
            return 0
        
        left = check(root.left)
        right = check(root.right)
        
        # Return -1 if unbalanced
        if left == -1 or right == -1 or abs(left - right) > 1:
            return -1
        return max(left, right) + 1
    
    return check(root) != -1
