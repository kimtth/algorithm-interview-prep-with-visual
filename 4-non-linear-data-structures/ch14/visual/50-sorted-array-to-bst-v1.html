Convert a sorted array into a height-balanced BST using divide and conquer. Pick the middle element as root, recursively build left and right subtrees.

def sortedArrayToBST(nums: List[int]) -> TreeNode:
    if not nums:
        return None

    mid = len(nums) // 2

    # Î∂ÑÌï† Ï†ïÎ≥µÏúºÎ°ú Ïù¥ÏßÑ Í≤ÄÏÉâ Í≤∞Í≥º Ìä∏Î¶¨ Íµ¨ÏÑ±
    node = TreeNode(nums[mid])
    node.left = sortedArrayToBST(nums[:mid])
    node.right = sortedArrayToBST(nums[mid + 1:])

    return node
